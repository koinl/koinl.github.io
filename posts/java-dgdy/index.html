使用递归算法求解累加和

class CumSum {
    int sum;
    int cumsum(int n) {
    if(n==0)
        sum=0;
        else sum=n+cumsum(n-1);
        return sum;
    }
}

使用递归算法求解累乘积

class CumProd {
    int p;
    int cumprod(int n) {
        if(n==0)
            p=1;
        else p=n*cumprod(n-1);
        return p;
    }
}
