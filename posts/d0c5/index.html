示例程序1:Fibonacci数列的前38个数.
解:package p_name;
　public class c_name {
　 　public static void main(String args[]) {
　　 　 　FibonacciRecursion f=new FibonacciRecursion();
　　 　 　for(int i = 1;i<=38/2;i++)
　　 　 　{
　　　　　　　System.out.println(f.fiborecur(i));
　　 　 　}
　 　}
}
class FibonacciRecursion
　　 　 　{
　　　　　　　int fiborecur(int n)
　　　　　　　{
　　　　　　 　　　if(n==1) return 1;
　　　　　　 　　　else if(n==2) return 1;
　　　　　　 　　　else return fiborecur(n-1)+fiborecur(n-2);
　　　　　　　}
　　 　 　}
